/* 3. zadatak */
#include <stdio.h>

long long zbir_interval(int a, int b) {
    long long sum = 0;
    if (a <= b) {
        for (int i = a; i <= b; ++i) sum += i;
    } else {
        for (int i = b; i <= a; ++i) sum += i;
    }
    return sum;
}

int main() {
    int k, n;
    printf("Unesite k i n: ");
    if (scanf("%d %d", &k, &n) != 2) return 0;
    long long s = zbir_interval(k, n);
    printf("Zbir svih brojeva izmedju %d i %d je: %lld\n", k, n, s);
    return 0;
}

    double r = d.poluprecnik;
    double h = d.duzina;
    return M_PI * r * r * h;
}
double bocna_povrsina_cm2(Drvo d) {

#include <stdio.h>
#include <stdlib.h>
#include <time.h> // Potrebno za srand(time(NULL))

// Iako je nazvano "Radnik" u svesci, radi se o oceni za Učenika
struct Radnik { 
    int srpski;
    int matematika;
    int fizika;
};

int main() {
    // Inicijalizacija generatora slučajnih brojeva
    srand(time(NULL)); 
    
    // Lista od 10 struktura (učenika)
    struct Radnik Lista[10]; 
    int i, najbolji = 0;
    float prosek, max = 0;
    
    // 1. GENERISANJE OCENA (od 1 do 5)
    for (i = 0; i < 10; i++) {
        Lista[i].srpski = rand() % 5 + 1;
        Lista[i].matematika = rand() % 5 + 1;
        Lista[i].fizika = rand() % 5 + 1;
    }

    // 2. ISPIS OCENA I PRONALAŽENJE NAJBOLJEG PROSEKA
    for (i = 0; i < 10; i++) {
        // Ispis ocena
        printf("Ucenik %d. Srpski = %d. Matematika = %d. Fizika = %d\n", 
               i + 1, Lista[i].srpski, Lista[i].matematika, Lista[i].fizika);
        
        // Racunanje proseka. Eksplicitna konverzija u float je ključna.
        prosek = ((float)Lista[i].srpski + Lista[i].matematika + Lista[i].fizika) / 3.0;
        
        // Provera da li je trenutni prosek veći od trenutnog maksimuma
        if (prosek > max) {
            max = prosek;
            najbolji = i + 1; // Čuvamo redni broj učenika (indeks + 1)
        }
    }
    
    // 3. ISPIS REZULTATA
    printf("\nNajbolji prosek: %.2f (Ucenik %d)\n", max, najbolji);
    
    return 0;
}


#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <list> // Neophodno za std::list
#include <iterator> // Neophodno za std::advance

using namespace std;

// Struktura za tačku sa X i Y koordinatom
struct tacka {
    int x;
    int x, y;
};

int main() {
    // Deklaracija STL liste koja sadrži strukture 'tacka'
    list<tacka> tacke; 
    list<tacka> cudo; // Dodatna lista
    tacka element;
    int i, z = 1;
    
    // 1. PUNJENJE LISTE SA 10 TAČAKA
    for (i = 0; i < 10; i++) {
        element.x = i * 10;
        element.y = i * 20;
        tacke.push_back(element); // Dodaje element na kraj liste
    }
    
    printf("Sadrzaj Liste:\n");
    
    // Iteratori za listu 'tacke'
    list<tacka>::iterator r; 
    
    // ITERACIJA 1: Ispis svih tačaka
    for (r = tacke.begin(); r != tacke.end(); r++) {
        element = *r; // Dereferenciranje iteratora da dobijemo element
        printf("%d. Tacka liste ima koordinate: x=%d / y=%d\n", 
               z++, element.x, element.y);
    }
    z = 1; // Resetovanje brojača

    // 2. MANIPULACIJA ITERATORIMA I ELEMENTIMA
    
    // Postavljanje iteratora 'r' na početak
    r = tacke.begin(); 
    
    // Dohvatanje tačke na poziciji 0
    element = *r; 
    printf("\nTačka na poziciji 0 je x=%d / y=%d\n", element.x, element.y);
    
    // Pomeranje iteratora 'r' za 2 mesta unapred
    advance(r, 2); 
    
    // Tačka na poziciji na koju pokazuje 'r' (pozicija 2)
    tacke.erase(r); // Brisanje tačke na poziciji 2

    // 3. KOPIRANJE (nekompletno na slici, ali logično prošireno)
    
    // Postavljanje iteratora 'r' ponovo na početak
    r = tacke.begin(); 
    
    // Kopiranje prve tačke iz liste 'tacke' u listu 'cudo'
    cudo.push_back(*r); 
    
    // 4. ITERACIJA 2: Ispis tačaka nakon brisanja
    printf("\nSadrzaj LISTE nakon brisanja:\n");
    z = 1;
    for (r = tacke.begin(); r != tacke.end(); r++) {
        element = *r;
        printf("%d. Tacka liste ima koordinate: x=%d / y=%d\n", 
               z++, element.x, element.y);
    }
    
    // 5. ISPIS LISTE 'cudo'
    printf("\nSadrzaj LISTE 'cudo':\n");
    list<tacka>::iterator w; 
    z = 1;
    for (w = cudo.begin(); w != cudo.end(); w++) {
        element = *w;
        printf("%d. Tacka liste 'cudo' ima koordinate: x=%d / y=%d\n", 
               z++, element.x, element.y);
    }
    return 0; }
