/* 3. zadatak */
#include <stdio.h>

long long zbir_interval(int a, int b) {
    long long sum = 0;
    if (a <= b) {
        for (int i = a; i <= b; ++i) sum += i;
    } else {
        for (int i = b; i <= a; ++i) sum += i;
    }
    return sum;
}

int main() {
    int k, n;
    printf("Unesite k i n: ");
    if (scanf("%d %d", &k, &n) != 2) return 0;
    long long s = zbir_interval(k, n);
    printf("Zbir svih brojeva izmedju %d i %d je: %lld\n", k, n, s);
    return 0;
}

#include <stdlib.h>
#include <time.h>

    double poluprecnik; // cm

} Drvo;

double zapremina_cm3(Drvo d) {
    double r = d.poluprecnik;
    double h = d.duzina;
    return M_PI * r * r * h;
}
double bocna_povrsina_cm2(Drvo d) {

Izvedene klase su krug i pravougaonik koje računaju obim i površinu.


        class kuca
        {
        public:
        virtual void ucitavanje() = 0;
        virtual float obim() = 0;
        virtual float povrsina() = 0;
        };

        class krug : public kuca
        {
        private:
        float r;
        public:
        void ucitavanje()
        {
        printf("Unesi poluprecnik:\n");
        scanf("%f", &r);
        }

        float obim()
        {
        return 2 * 3.14 * r;
        }

        float povrsina()
        {
        return 3.14 * r * r;
        }
        };

        class pravougaonik : public kuca
        {
        private:
        float a, b;
        public:
        void ucitavanje()
        {
        printf("Unesi stranicu a:\n");
        scanf("%f", &a);
        printf("Unesi stranicu b:\n");
        scanf("%f", &b);
        }

        float obim()
        {
        return 2 * (a + b);
        }

        float povrsina()
        {
        return a * b;
        }
        };

        int main()
        {
        int p;
        printf("Izaberi oblik:\n");
        printf("1 za krug\n");
        printf("2 za pravougaonik\n");
        scanf("%d", &p);
        if (p == 1)
        {
        krug x;
        x.ucitavanje();
        printf("Obim je: %10.2f\n", x.obim());
        printf("Povrsina je: %10.2f\n", x.povrsina());
        }
        if (p == 2)
        {
        pravougaonik x;
        x.ucitavanje();
        printf("Obim je: %10.2f\n", x.obim());
        printf("Povrsina je: %10.2f\n", x.povrsina());
        }
        return 0;
        }
    
Napravi interfejs oblik koji definiše funkciju za unos podataka i funkciju za unos površine. Zatim napraviti klasu pravougaonik koja implementira interfejs.

UU main funkciji koristiti pokazivač na interfejs, dinamički kreirati klasu pravougaonik, uneti podatke i ispisati površinu.


        class oblik{
        public:
        virtual void ucitavanje()=0;
        virtual float povrsina()=0;
        virtual ~oblik() {} //mora da ima konstruktor
        };

        class pravougaonik:public oblik{
        private:
        float sirina,duzina;
        public:
        void ucitavanje() override
        {
        printf("Unesite sirinu:\n");
        scanf("%f",&sirina);
        printf("Unesite duzinu:\n");
        scanf("%f",&duzina);
        }
        float povrsina()override
        {
        return duzina*sirina;
        }
        };

        int main()
        {
        //ovo * je pokazivac na interfejs
        oblik* x=new pravougaonik();
        //ovo new pravi dinamicki novi objekat
        x->ucitavanje();
        //ucitavanje preko pokazivaca
        printf("Povrsina je: %10.2f\n",x->povrsina());
        delete x;
        //obavezno oslobadjanje memorije
        return 0;
        }
